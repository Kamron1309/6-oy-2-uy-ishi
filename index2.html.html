      <h3>{name}</h3>
      <p>Yoshi: {age}</p>

    </div>
  );
}

Mapping

Ro'yxat ma'lumotlarini aylantirib, ko'p elementlar yaratish uchun ishlatiladi.

const users = [
  { id: 1, name: 'Ali', age: 25 },
  { id: 2, name: 'Vali', age: 30 },
  { id: 3, name: 'Hasan', age: 28 }
];

return (
  <ul>
    {users.map(user => (
      <li key={user.id}>
        {user.name} - {user.age} yosh
      </li>
    ))}
  </ul>
);
